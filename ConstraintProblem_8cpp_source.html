00095         ++row;
00096       }     
00097     }    
00098     lp.aliasingRows_ = (row - 1) - lp.structuralRows_;
00099     for (vector<Constraint>::const_iterator c = constraints_.begin(); c != constraints_.end(); ++c) {
00100       if (c->GetType() == Constraint::NORMAL) {
00101         c->AddToLPP(lp.lp_, row, lp.varToCol_);
00102         ++row;
00103       }     
00104     }    
00105     lp.realRows_ = row - lp.structuralRows_ - 1;
00106   }
00107 
00108   for (size_t i = 1; i <= vars.size(); ++i) {
00109     glp_set_col_bnds(lp.lp_, i, GLP_FR, 0.0, 0.0);

00111

00113     // Set objective coeficients
